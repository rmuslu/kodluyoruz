1)SELECT title, length 
FROM film 
WHERE title ILIKE '%n' 
ORDER BY length DESC 
LIMIT 5 ;

2)SELECT title, length 
FROM film 
WHERE title ILIKE '%n' 
ORDER BY length ASC 
OFFSET 1
LIMIT 1 ;

WHERE store_id = 1
